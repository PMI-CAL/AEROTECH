;TOOL #1 

	;OP SUB START CODE TO INIT
	PSOCONTROL	X RESET
	PSOPULSE X	TIME ($PULSEWIDTHSCRIBE + 1) $PULSEWIDTHSCRIBE CYCLES 1
	PSOOUTPUT X PULSE
	PSOTRACK X	INPUT 2,1
	PSODISTANCE X FIXED $SCRIBEGAP
	SIDEON
	
	;OP SUB END CODE TO RESET
	PSOCONTROL	X RESET
	SIDEOFF
	G0
	
;TOOL #2
	;OP SUB START CODE TO INIT
	PSOCONTROL	X RESET
	PSOPULSE X	TIME ($PULSEWIDTHSCRIBE2 +	1) $PULSEWIDTHSCRIBE2 CYCLES 1
	PSOOUTPUT X PULSE
	PSOTRACK X	INPUT 2,1
	PSODISTANCE X FIXED $SCRIBEGAP2
	COAXON	

	;OP SUB END CODE TO RESET
	PSOCONTROL	X RESET
	COAXOFF
	G0

;TOOL #4
	;MOVE THESE TWO HOLE VARIABLES INTO N2 SUBROUTINE PRIOR TO OP SUB CALL
	$HOLERADIUS 	= 0.000
	$HOLELEADINCUT 	= 0.000
	
	;OP SUB START CODE TO INIT
	PSOCONTROL	X RESET
	PSOPULSE X	TIME $PULSEPERIODHOLE $PULSEWIDTHHOLE
	CURRENT1ON
	COAXON
	$XOFFSET=$XOFFSET+($HOLERADIUS-$HOLELEADINCUT)
	CALL INITIALIZE

	;OP SUB END CODE TO RESET
	$XOFFSET=$XOFFSET-($HOLERADIUS-$HOLELEADINCUT)
	CALL INITIALIZE
	CURRENT1OFF
	PSOCONTROL	X RESET
	COAXOFF
	G0
